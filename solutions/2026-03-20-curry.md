export default function curry(func) {
  //define a helper function to collect arguments
  return function curried(...args) {
    //check if the number of args privied is enough
    if (args.length >= func.length) {
      //base case: enough args, excute the orignal function
      return func.apply(this, args);
    }

    //recursive case: not enough, return new function to collect more
    //bind creates a new function, this point the original one.
    return curried.bind(this, ...args);
  };
}

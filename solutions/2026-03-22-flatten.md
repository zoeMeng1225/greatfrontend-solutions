/**
 * @param {Array<*|Array>} value
 * @return {Array}
 */
export default function flatten(value) {
  //initialize an empty array to store result
  let result = [];

  //iterate through each elements in the array
  value.forEach((item) => {
    if (Array.isArray(item)) {
      //if it is an array, recursively flatten it and merge init result
      result = result.concat(flatten(item));
    } else {
      //if it not an array, just push the value inio result
      result.push(item);
    }
  });

  return result;
}

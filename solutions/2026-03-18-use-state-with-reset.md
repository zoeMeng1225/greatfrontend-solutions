import { useState, useRef, useCallback } from "react";

export default function useStateWithReset(initialStateOrInitializer) {
  //use useState to manage the current value
  const [state, setState] = useState(initialStateOrInitializer);

  //use a ref to lock the very first value as our reset target;
  const initialStateRef = useRef(state);

  //the reset function sets the state back to the stored initial value
  const reset = useCallback(() => {
    setState(initialStateRef.current);
  });

  //return the valua, setter, and reset function as an array;
  return [state, setState, reset];
}

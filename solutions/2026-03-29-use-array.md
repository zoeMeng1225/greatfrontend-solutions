import { useState, useCallback } from "react";

export default function useArray(defaultValue) {
  const [array, setArray] = useState(defaultValue);

  //push : add an item to the end
  const push = useCallback((item) => {
    setArray((a) => [...a, item]);
  });

  //update: replace item at a specific index
  const update = useCallback((index, newItem) => {
    setArray((a) => {
      const copy = [...a];
      copy[index] = newItem;
      return copy;
    });
  }, []);

  //remove: delete item at a specific index
  const remove = useCallback((index) => {
    setArray((a) => a.filter((_, i) => i !== index));
  }, []);

  //filter: keep items that match the callback condition
  const filter = useCallback((callback) => {
    setArray((a) => a.filter(callback));
  }, []);

  //set: replace the entire array
  const set = useCallback((newArray) => {
    setArray(newArray);
  }, []);

  //clear: empty the array
  const clear = useCallback(() => {
    setArray([]);
  }, []);

  return { array, push, update, remove, filter, set, clear };
}

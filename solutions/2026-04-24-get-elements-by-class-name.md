/**
 * @param {Element} element
 * @param {string} classNames
 * @return {Array<Element>}
 */
export default function getElementsByClassName(element, classNames) {
  //edge case: if the classname is empty or just space return empty array
  const trimed = classNames.trim();
  if (!trimed) return [];

  //if the classname is not empty, we convert string to an array of class name
  //handling multiple spaces.
  const targets = trimed.split(/\s+/);
  const results = [];

  //helper function: Depth-first search(DFS)
  function traverse(node) {
    //Iterate throght all direct child elements(element.children excludes text node)
    for (const child of node.children) {
      //check if this child has every target class
      const hasAllClasses = targets.every((cls) =>
        child.classList.contains(cls),
      );

      if (hasAllClasses) {
        results.push(child);
      }

      //recursively traverse the descendant of this child
      traverse(child);
    }
  }
  //start traversal from the children of the provided element(excludes the element itself)
  traverse(element);

  return results;
}

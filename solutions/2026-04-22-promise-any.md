/**
 * @param {Array} iterable
 * @return {Promise}
 */
export default function promiseAny(iterable) {
  return new Promise((resolve, reject) => {
    //if the array is empty, reject immediately
    if (iterable.length === 0) {
      return reject(new AggregateError([], "All promises were rejected"));
    }

    let rejectedCounted = 0;
    const errors = new Array(iterable.length);

    iterable.forEach((item, index) => {
      //wrap item in Promise.resolve to handle non-promise values
      Promise.resolve(item).then(
        (value) => {
          //success: if any promise fulfills, resolve the whole thing
          resolve(value);
        },
        (error) => {
          //failure: keep truck of error and theirs and orignal positions
          errors[index] = error;
          rejectedCounted++;

          //if all promises fail, reject with AggregateError
          if (rejectedCounted === iterable.length) {
            reject(new AggregateError(errors, "All Promises were rejected"));
          }
        },
      );
    });
  });
}

import { useState, useEffect, useRef } from "react";

type CheckboxItem = {
  id: number;
  name: string;
  checked: boolean | "indeterminate";
  children?: CheckboxItem[];
};

//Recursive sub-component
function CheckboxNode({
  node,
  onToggle,
}: {
  node: CheckboxItem;
  onToggle: (id: number) => void;
}) {
  const inputRef = useRef<HTMLInputElement>(null);

  //indeterminate is a DOM property, not an HTML attribute
  useEffect(() => {
    if (inputRef.current) {
      inputRef.current.indeterminate = node.checked === "indeterminate";
    }
  }, [node.checked]);

  return (
    <div style={{ marginLeft: "20px", textAlign: "left", marginTop: "8px" }}>
      <label
        style={{ cursor: "pointer", display: "flex", alignItems: "center" }}
      >
        <input
          ref={inputRef}
          type="checkbox"
          checked={node.checked === true}
          onChange={() => onToggle(node.id)}
        />
        <span style={{ marginLeft: "8px" }}>{node.name}</span>
      </label>
      {/* if node has children, render them recursively*/}
      {node.children && (
        <div>
          {node.children.map((child) => (
            <CheckboxNode key={child.id} node={child} onToggle={onToggle} />
          ))}
        </div>
      )}
    </div>
  );
}

export default function Checkboxes({
  defaultCheckboxData,
}: Readonly<{
  defaultCheckboxData: CheckboxItem[];
}>) {
  //use state to manage the tree structure
  const [data, setData] = useState<CheckboxItem[]>(defaultCheckboxData);

  //helper: update all descendants(downwards)
  const updateDescendants = (
    nodes: CheckboxItem[],
    isChecked: boolean,
  ): CheckboxItem[] => {
    return nodes.map((node) => ({
      ...node,
      checked: isChecked,
      children: node.children
        ? updateDescendants(node.children, isChecked)
        : undefined,
    }));
  };

  //helper: Recursive toggle and recalculate(bottom-up)
  const toggleRecursive = (
    nodes: CheckboxItem[],
    targetId: number,
  ): CheckboxItem[] => {
    return nodes.map((node) => {
      //found the target node
      if (node.id === targetId) {
        const nextValue = node.checked !== true; //toggle logic
        return {
          ...node,
          checked: nextValue,
          children: node.children
            ? updateDescendants(node.children, nextValue)
            : undefined,
        };
      }

      //if not the target, but maybe in children
      if (node.children) {
        const updatedChildren = toggleRecursive(node.children, targetId);

        //re-calculate parent state
        const allChecked = updatedChildren.every((c) => c.checked === true);
        const allUnchecked = updatedChildren.every((c) => c.checked === false);

        let newState: boolean | "indeterminate" = "indeterminate";
        if (allChecked) newState = true;
        else if (allUnchecked) newState = false;

        return {
          ...node,
          checked: newState,
          children: updatedChildren,
        };
      }
      return node;
    });
  };

  const handleToggle = (id: number) => {
    setData((prev) => toggleRecursive(prev, id));
  };

  return (
    <div>
      <h1>Nested Cheeckboxes</h1>
      <div>
        {data.map((item) => (
          <CheckboxNode key={item.id} node={item} onToggle={handleToggle} />
        ))}
      </div>
    </div>
  );
}

/**
 * @param {(...args: Array<unknown>) => unknown} func
 * @param {number} wait
 * @returns {(...args: Array<unknown>) => void}
 */
export default function debounce(func, wait) {
  //initialize a timer veriable, this create a closure to track timer status
  let timer = null;

  //return wrapper function with rest operator to catch all arguments
  return function (...args) {
    //check if the timer is running, which means user is typing, we need to clear the timer
    if (timer) {
      clearTimeout(timer);
    }

    //set a new timer
    timer = setTimeout(() => {
      //excute the orignal function with corrent context(this) and argument
      func.apply(this, args);
    }, wait);
  };
}

Array.prototype.myFilter = function (callbackFn, thisArg) {
  const arr = this;
  let result = [];

  for (let i = 0; i < arr.length; i++) {
    if (callbackFn.call(thisArg, arr[i], i, arr)) {
      if (i in arr) {

};

/**
 * @param {...(any|Object|Array<any|Object|Array>|Function)} args
 * @return {string}
 */
export default function classNames(...args) {
  const classes = new Set<string>();

  function process(item: any) {
    if (!item) return;
    const type = typeof item;
    if (type === "string" || type === "number") {
      classes.add(String(item));
    } else if (type === "function") {
      process(item());
      //1: item():动作：你把壳剥开（执行 item()）
      //2process(...) —— “把拿到的东西再去过一遍安检”(递归)
    } else if (Array.isArray(item)) {
      item.forEach(process);
      //item.forEach(element => process(element))
    } else if (type === "object") {
      //遍历所有的key,比如 'foo', 'bar'）
      Object.keys(item).forEach((key) => {
        //检查这个key 对应的值是true 还是false
        if (item[key]) {
          classes.add(key);
        } else {
          classes.delete(key);
        }
      });
    }
  }
  //遍历所有的参数,调用 process 函数
  args.forEach((ele) => process(ele));
  return Array.from(classes).join(' ')
}

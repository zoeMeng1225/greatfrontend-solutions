  //first, i'll create a timer veriable. this creates closure to keep track of the timer state
  let timer = null;

  //the debounced function returns a new wrapper function. I'll use rest operater to catch all arguments
  return function (...args) {
    //i 'll save the current context to make sure we don't lose the original 'this'
    const context = this;

    //here is core logicm if a timer is already running, which means the user us still typing, we cancel it
    if (timer) {
      clearTimeout(timer);
    }

    //then, we set a brand new timer
    timer = setTimeout(() => {
      //if the user stops for wait millsecond, we finally execute the original function with thr correct and arguments
      func.apply(context, args);
    }, wait);
  };
}

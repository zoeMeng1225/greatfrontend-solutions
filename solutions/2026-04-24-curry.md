/**
 * @param {(...args: Array<unknown>) => unknown} func
 * @returns {(...args: Array<unknown>) => unknown}
 */
export default function curry(func) {
  //define a helper function to collect arguments
  return function curried(...args) {
    //check if the number of args provided is enough
    if (args.length >= func.length) {
      //base case: enough args, execute the orignal function
      return func.apply(this, args);
    }

    //if not enought: return new function to collect more
    //bind create a new function, this point the original one
    return curried.bind(this, ...args);
  };
}

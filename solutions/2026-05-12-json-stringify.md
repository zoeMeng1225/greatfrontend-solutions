/**
 * @param {unknown} value
 * @returns {string}
 */
export default function jsonStringify(value) {
  //handle base case: null
  if (value === null) {
    return "null";
  }

  //handle base case : string(need to be wrapped in double quotes)
  if (typeof value === "string") {
    return `"${value}"`;
  }

  //handle base case: number and boolean
  if (typeof value === "boolean" || typeof value === "number") {
    return String(value);
  }

  //handle recursive case : array
  if (Array.isArray(value)) {
    //recursively stringify each element in the array
    const arrayItems = value.map((item) => jsonStringify(item));
    return `[${arrayItems.join(",")}]`;
  }

  //handle recursive case: Object(plain objects)
  if (typeof value === "object") {
    const objectKeys = Object.keys(value);

    const objectItems = objectKeys.map((key) => {
      //recursively stringify the value
      const stringifiedValue = jsonStringify(value[key]);

      //Object keys in JSON must be wrapped in double quotes
      return `"${key}":${stringifiedValue}`;
    });

    return `{${objectItems.join(",")}}`;
  }
}

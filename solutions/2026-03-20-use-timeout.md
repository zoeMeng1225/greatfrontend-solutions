/**
 * @param {() => void} callback
 * @param {number | null} delay
 */

import { useRef, useEffect } from "react";

export default function useTimeout(callback, delay) {
  //use useRef to store newest callback
  const savedCallback = useRef(callback);

  //remember the latest callback if it changes
  //this ensures the timer always calls the newest version without resttring
  useEffect(() => {
    savedCallback.current = callback;
  }, [callback]);

  //set up the timeout logic
  useEffect(() => {
    //if delay is null, do not set a timer
    if (delay === null) return;

    //execute the saved callback after the delay
    const id = setTimeout(() => {
      savedCallback.current();
    }, delay);

    //cleanup function: clear the timerout if delay changes or component upmount
    return () => clearTimeout(id);
  }, [delay]); //only restart if delay changes
}

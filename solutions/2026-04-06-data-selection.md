/**
 * @param {Array<{user: number, duration: number, equipment: Array<string>}>} sessions
 * @param {{user?: number, minDuration?: number, equipment?: Array<string>, merge?: boolean}} [options]
 * @return {Array}
 */
export default function selectData(sessions, options) {
  //先copy,adviod to change orianal data
  let data = sessions.map((s) => ({
    ...s,
    equipment: [...s.equipment],
  }));

  if(!options) return data

  //first check with merge
  //if need to merge. according to user to combime
  if (options.merge) {
    const map = new Map(); //user => mergedSession
    const lastIndex = new Map(); //user => last appear的index

    data.forEach((s, index) => {
      //如果没有
      if (!map.has(s.user)) {
        map.set(s.user, {...s, equipment: [...s.equipment] });
      } else {
        //有
        const m = map.get(s.user);
        m.duration += s.duration;
        m.equipment = Array.from(
          new Set([...m.equipment, ...s.equipment]),
        ).sort(); //去重+排序
      }
      lastIndex.set(s.user, index);
    })
  

    //变成数组， 并按lastIndex 排序，保证位置在「最后一次出现」那
    data = Array.from(map.values()).sort(
      (a, b) => lastIndex.get(a.user) - lastIndex.get(b.user)
    )}

    //按user过滤
    if(options.user !== undefined){
      data = data.filter(s => s.user === options.user)
    }

    //按minDuration filter
    if(options.minDuration !== undefined){
      data = data.filter(s => s.duration >= options.minDuration)
      console.log(data)
    } 

    //按equipment filter
    if(options.equipment && options.equipment.length){
      const set = new Set(options.equipment);
      data = data.filter(s => s.equipment.some(eq => set.has(eq)))
    }
  

  return data;
}

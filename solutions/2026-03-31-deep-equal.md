/**
 * @param {*} valueA
 * @param {*} valueB
 * @return {boolean}
 */
export default function deepEqual(valueA, valueB) {
  //primitive equality or same reference
  if (valueA === valueB) return true;

  //if either is not an object(or is null), they can't be deep equal
  if (
    valueA === null ||
    valueB === null ||
    typeof valueA !== "object" ||
    typeof valueB !== "object"
  ) {
    return false;
  }

  //handle arrays sparately
  const isArrayA = Array.isArray(valueA);
  const isArrayB = Array.isArray(valueB);

  if (isArrayA !== isArrayB) return false;

  if (isArrayA && isArrayB) {
    if (valueA.length !== valueB.length) return false;
    //recursively check every element
    for (let i = 0; i < valueA.length; i++) {
      if (!deepEqual(valueA[i], valueB[i])) return false;
    }
    return true;
  }

  //handle objects
  const keysA = Object.keys(valueA);
  const keysB = Object.keys(valueB);

  if (keysA.length !== keysB.length) return false;

  for (const key of keysA) {
    //Check if key exists in B and recurse
    if (
      !Object.prototype.hasOwnProperty.call(valueB, key) ||
      !deepEqual(valueA[key], valueB[key])
    ) {
      return false;
    }
  }

  return true;
}

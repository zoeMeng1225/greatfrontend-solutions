/**
 * @param {Function} func
 * @returns Function
 */
export default function memoize(func) {
  //use a Map instead of a plain object
  const cache = new Map();

  return function (arg) {
    //use Map.has() to check if the key exists(preserves the orignal type of arg)
    if (cache.has(arg)) {
      //use Map.get() to retrieve the value
      return cache.get(arg);
    }
    //execute original function
    const result = func.apply(this, [arg]);

    // use Map.set() to store the result
    cache.set(arg, result);

    return result;
  };
}

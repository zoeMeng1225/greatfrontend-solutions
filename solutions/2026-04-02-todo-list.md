```javascript
import { useState } from "react";

export default function App() {
  //initialize state for the list and the current input value
  const [todos, setTodo] = useState([
    { id: 1, text: "Walk the dog" },
    { id: 2, text: "Water the plants" },
    { id: 3, text: "Wash the dishes" },
  ]);

  const [input, setInput] = useState("");

  //make a function to add a new task
  const addTask = (e) => {
    e.preventDefault();
    //trim white space and prevent empty task
    if (input.trim() === "") return;

    const newTodo = {
      id: Date.now(), //use timestamp as a unique id
      text: input,
      completed: false, //initial status is false
    };

    setTodo([...todos, newTodo]); //update the list
    setInput(""); //clear input field
  };

  //toggle completion status
  const toggleTodo = (id) => {
    //map through the list and filp completed flag for matching id
    const updatedTodo = todos.map((todo) =>
      todo.id === id ? { ...todo, completed: !todo.completed } : todo,
    );

    setTodo(updatedTodo);
  };

  //deleteTask method: delete a task by id
  const removeTask = (id) => {
    //use filter, to filter out the task with the matching id
    setTodo(todos.filter((todo) => todo.id !== id));
  };

  //add task with enter key(keydown event)
  const handleKeyDown = (e) => {
    //check if the pressed key is enter
    if (e.key === "Enter") {
      addTask();
    }
  };

  return (
    <div>
      <h1>Todo List</h1>
      <div>
        <input
          type="text"
          placeholder="Add your task"
          value={input}
          onKeyDown={handleKeyDown}
          onChange={(e) => setInput(e.target.value)}
        />
        <div>
          <button onClick={addTask}>Add</button>
        </div>
      </div>
      <ul>
        {todos.map((todo) => (
          <li key={todo.id}>
            <span
              onClick={() => toggleTodo(todo.id)}
              style={{
                color: todo.completed ? "gray" : "black",
                cursor: "pointer",
                textDecoration: todo.completed ? "line-through" : "none",
              }}
            >
              {todo.text}
            </span>
            <button onClick={() => removeTask(todo.id)}>delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

```

```javascript
import { useState } from "react";

export default function App() {
  //initialize state for the list and the current input value
  const [todos, setTodos] = useState([
    { id: 1, text: "Walk the dog" },
    { id: 2, text: "Water the plants" },
    { id: 3, text: "Wash the dishes" },
  ]);

  const [input, setInput] = useState("");

  //the function for add a new task
  const addTasks = (e) => {
    e.preventDefault();
    //trim white space and prevent task
    if (input.trim() === "") return;

    const newTodo = {
      id: Date.now(), //use timestamp as a unique id
      text: input,
      completed: false, //initial status is false
    };

    setTodos([...todos, newTodo]); //update the list
    setInput(""); //clear input field
  };

  //delete todo method: delete a task by id
  const removeTask = (id) => {
    setTodos(todos.filter((todo) => todo.id !== id));
  };

  //add tack with enter key(key down event)
  const handleKeyDown = (e) => {
    //check if the passed key is enter
    if (e.key === "Enter") {
      addTasks(e);
    }
  };

  const toggleComplete = (id) => {
    const updatedTodo = todos.map((todo) =>
      todo.id === id ? { ...todo, completed: !todo.completed } : todo,
    );

    setTodos(updatedTodo);
  };

  return (
    <div>
      <h1>Todo List</h1>
      <div>
        <input
          type="text"
          placeholder="Add your task"
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={handleKeyDown}
        />
        <div>
          <button onClick={addTasks}>Submit</button>
        </div>
      </div>
      <ul>
        {todos.map((todo) => (
          <li key={todo.id}>
            <span
              style={{
                cursor: "pointer",
                textDecoration: todo.completed ? "line-through" : "none",
                color: todo.completed ? "gray" : "black",
              }}
            >
              {todo.text}
            </span>
            <button
              onClick={() => toggleComplete(todo.id)}
              style={{ display: todo.completed ? "none" : "inline-block" }}
            >
              Complete
            </button>
            <button onClick={() => removeTask(todo.id)}>Delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

```

import { useState, useEffect } from "react";
/**
 * @template T
 * @param {() => Promise<T>} fn
 * @param {import("react").DependencyList} deps
 */
export default function useQuery(fn, deps = []) {
  //initialize the state object, default status is loading
  const [state, setState] = useState({
    status: "loading",
  });

  useEffect(() => {
    //"ignore" flag is crucial for avoilding race conditions
    let ignore = false;

    //reset state to 'loading'
    setState({ status: "loading" });

    //execute the promise-returning function
    fn()
      .then((data) => {
        //only update state if this is still the most recent request
        if (!ignore) {
          setState({ status: "success", data });
        }
      })
      .catch((error) => {
        //only update state if this is still the most recent request
        if (!ignore) {
          setState({ status: "error", error });
        }
      });

    //cleanup function runs when deps change or the component unmounts
    return () => {
      ignore = true;
    };
    //disable exhaustive-deps rule here as we specifically
    //want to spread the user's deps array
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, deps);

  //return current state object;
  return state;
}

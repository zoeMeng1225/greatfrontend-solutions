/** @typedef {string | HTMLElementNode} HTMLNode */

/**
 * @param {HTMLElementNode} root
 * @return {string}
 */
export default function serializeHTML(root) {
  //define a recursive helper function taking the node and current depth
  const serializeNode = (node, depth) => {
    //generate the correct indentation based on depth ussing tabs
    const indent = "\t".repeat(depth);

    //base case: if the node is plain string, return it with indentation
    if (typeof node === "string") {
      return `${indent}${node}`;
    }

    //prepare the opening and closing tags with current indentation
    const openTag = `${indent}<${node.tag}>`;
    const closeTag = `${indent}</${node.tag}>`;

    //edge case: if there are no children, return just the tag
    if (!node.children || node.children.length === 0) {
      return `${openTag}\n${closeTag}`;
    }

    //recursive step: map through all children, increasing the depth by 1
    const serializedChildren = node.children
      .map((child) => serializeNode(child, depth + 1))
      .join("\n");

    //combine the opening tag, the serialied children, and the closing tag
    return `${openTag}\n${serializedChildren}\n${closeTag}`;
  };

  //start the recursion from the root node at depth 0
  return serializeNode(root, 0);
}

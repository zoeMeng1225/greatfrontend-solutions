export default function squashObject(obj) {
  //initialize the result to squash
  const result = {};

  //define the DFS helper function
  function dfs(currentVal, path) {
    //base case: if it is a primitive value or strictly null
    if (typeof currentVal !== "object" || currentVal === null) {
      //only assign if the path is valid
      if (path !== "") {
        result[path] = currentVal;
      }

      return;
    }

    //recursive case for objects and arrays
    //Object.key() works for both objects and arrays(return stringified indices)
    const keys = Object.keys(currentVal);

    //Edge case: empty objects/arrays are usually ignored in flattenning

    for (const key of keys) {
      const newPath = [path, key].filter(Boolean).join(".");

      //recurse deeper
      dfs(currentVal[key], newPath);
    }
  }

  //start DFS with an empty path
  dfs(obj, "");
  return result;
}

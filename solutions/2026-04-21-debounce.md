/**
 * @param {(...args: Array<unknown>) => unknown} func
 * @param {number} wait
 * @returns {(...args: Array<unknown>) => void}
 */
export default function debounce(func, wait) {
  //First, i 'll create a timer veriable, this creates closure to keep track of the timer state
  let timer = null;

  //This function return a new wrapper funtion, I 'll use rest operator to catch all arguments
  return function (...args) {
    //current context to make sure we don't lost the original 'this'
    const context = this;

    //here is core logic if a timer is already running, which means the users still typing, and we cancel it
    if (timer) {
      clearTimeout(timer);
    }

    //then, we set a new timer
    timer = setTimeout(() => {
      //if the user stop for wait ms(millsecond), we execute the original function
      func.apply(context, args);
    }, wait);
  };
}
